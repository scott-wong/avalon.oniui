图片轮播组件-自定义宽高

自定义宽高1200×200
自定义宽高300×300 
<!DOCTYPE html>
<html>
    <head>
        <title>carousel组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../avalon.js"></script>
        <script src="../highlight/shCore.js"></script>
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css"/>
        <script>
            require(["carousel/avalon.carousel"], function() {
                avalon.define("demo1", function(vm) {
                    vm.carousel = {
                        pictures: [
                        "http://placehold.it/900x200/333/dddddd.jpg&text=1200-200",
                        "http://placehold.it/900x200/555/dddddd.jpg&text=1200-200",
                        "http://placehold.it/900x200/777/dddddd.jpg&text=1200-200"],
                        pictureWidth: 900, //@param pictureWidth 图片显示宽度
                        pictureHeight: 200 //@param pictureHeight 图片显示高度
                    }
                    vm.$skipArray = ["carousel"]
                })
                avalon.define("demo2", function(vm) {
                    vm.carousel = {
                        pictures: [
                        "http://placehold.it/300x300/333/dddddd.jpg&text=300-300",
                        "http://placehold.it/300x300/555/dddddd.jpg&text=300-300",
                        "http://placehold.it/300x300/777/dddddd.jpg&text=300-300"],
                        pictureWidth: 300, //@param pictureWidth 图片显示宽度
                        pictureHeight: 300 //@param pictureHeight 图片显示高度
                    }
                    vm.$skipArray = ["carousel"]
                })
                avalon.scan()
            })
        </script>
    </head>
    <body>
        <div class="wrapper">
        <h1>图片轮播组件-自定义宽高</h1>
        <fieldset ms-controller="demo1">
            <legend>自定义宽高1200×200</legend>
            <div ms-widget="carousel"></div>
        </fieldset>
        <fieldset ms-controller="demo2">
            <legend>自定义宽高300×300</legend>
            <div ms-widget="carousel"></div>
        </fieldset>
    </div>
    </body>
</html>
