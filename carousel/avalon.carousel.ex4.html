图片轮播组件-自定义effect

自定义effect-slide
自定义effect-fade
自定义effect-none 
<!DOCTYPE html>
<html>
    <head>
        <title>carousel组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../avalon.js"></script>
        <script src="../highlight/shCore.js"></script>
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css"/>
        <script>
            require(["carousel/avalon.carousel"], function() {
                avalon.define("demo1", function(vm) {
                    vm.carousel = {
                        pictures: [
                        "http://placehold.it/500x200/333333/dddddd.jpg&text=slide1", 
                        "http://placehold.it/500x200/555555/dddddd.jpg&text=slide2",
                        "http://placehold.it/500x200/777777/dddddd.jpg&text=slide3"],
                        effect: "slide" //@图片切换类型，默认为"slide"，取值：none:无特效 / fade:渐隐 / slide:滑动
                    }
                    vm.$skipArray = ["carousel"]
                })
                avalon.define("demo2", function(vm) {
                    vm.carousel = {
                        pictures: [
                        "http://placehold.it/500x200/33aa33/dddddd.jpg&text=fade1", 
                        "http://placehold.it/500x200/aa3333/dddddd.jpg&text=fade2",
                        "http://placehold.it/500x200/3333aa/dddddd.jpg&text=fade3"],
                        effect: "fade"
                    }
                    vm.$skipArray = ["carousel"]
                })
                avalon.define("demo3", function(vm) {
                    vm.carousel = {
                        pictures: [
                        "http://placehold.it/500x200/555555/dddddd.jpg&text=none1", 
                        "http://placehold.it/500x200/555555/dddddd.jpg&text=none2",
                        "http://placehold.it/500x200/555555/dddddd.jpg&text=none3"],
                        effect: "none"
                    }
                    vm.$skipArray = ["carousel"]
                })
                avalon.scan()
            })
        </script>
    </head>
    <body>
        <div class="wrapper">
        <h1>图片轮播组件-自定义effect</h1>
        <fieldset ms-controller="demo1">
            <legend>自定义effect-slide</legend>
            <div ms-widget="carousel"></div>
        </fieldset>
        <fieldset ms-controller="demo2">
            <legend>自定义effect-fade</legend>
            <div ms-widget="carousel"></div>
        </fieldset>
        <fieldset ms-controller="demo3">
            <legend>自定义effect-none</legend>
            <div ms-widget="carousel"></div>
        </fieldset>
    </div>
    </body>
</html>
